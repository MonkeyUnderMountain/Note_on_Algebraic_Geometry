\section{Formal Schemes}


    \begin{theorem}\label{thm:theorem_on_the_formal_functions}
        Let \(X\) be a proper scheme over a noetherian complete local ring \(A\) with maximal ideal \(\frakm\).
        Let \(\calF\) be a coherent sheaf on \(X\).
        For each \(n \geq 0\), let \(X_n = X \times_{\Spec A} \Spec A/\frakm^{n+1}\) and \(\calF_n = \calF  X_n\).
        Then the natural map
        \[
            H^p(X, \calF) \to \varprojlim_n H^p(X_n, \calF_n)
        \]
        is an isomorphism of \(A\)-modules for all \(p \geq 0\).
        \Yang{To be checked.}
    \end{theorem}
