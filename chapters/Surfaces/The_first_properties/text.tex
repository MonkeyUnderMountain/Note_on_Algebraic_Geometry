\section{The first properties of surfaces}

Let \(\kkk\) be an algebraically closed field of arbitrary characteristic.
Unless otherwise specified, all varieties are defined over \(\kkk\).

\subsection{Basic concepts}

    \begin{definition}\label{def:surface}
        A \emph{surface} is a two-dimensional integral scheme of finite type over an algebraically closed field \(\kkk\).
        A \emph{projective surface} is a surface that is projective over \(\kkk\).
        A \emph{smooth surface} is a surface that is smooth over \(\kkk\).
        \Yang{To be checked.}
    \end{definition}
